  13. 切片

    # 通过切片来获取指定的元素
# 语法：列表[起始:结束] 
#   通过切片获取元素时，会包括起始位置的元素，不会包括结束位置的元素
#   做切片操作时，总会返回一个新的列表，不会影响原来的列表
#   起始和结束位置的索引都可以省略不写
#   如果省略结束位置，则会一直截取到最后
#   如果省略起始位置，则会从第一个元素开始截取
#   如果起始位置和结束位置全部省略，则相当于创建了一个列表的副本
stus = [12,3,4,5,6,5,7,8]
print(stus[1:])
print(stus[:3])
print(stus[:])
print(stus)

# 语法：列表[起始:结束:步长] 
# 步长表示，每次获取元素的间隔，默认值是1
print(stus[0:5:3])
# 步长不能是0，但是可以是负数
# print(stus[::0]) ValueError: slice step cannot be zero
# 如果是负数，则会从列表的后部向前边取元素
print(stus[::-1])

  19. copy

    import copy
# copy() a与b的地址相同  a和b是一个东西
a=[1,2,3]
b=a
print(id(a),id(b))
print(a,b)
a[1]=22
print(a,b)
print(id(a)==id(b))

# copy.copy() a与b地址不同 a与b不是一个东西
c=copy.copy(a)
print(id(a),id(c))
print(a,c)
a[1]=22
print(a,c)
print(id(c)==id(c))
# 但a的深层与d的深层相同  a[2]与d[2]是一个东西
a = [1,2,[3,4]]
d = copy.copy(a)
print(id(a)==id(d))
print(id(a[2])==id(d[2]))

# copy.deepcopy() a与e完全不同
e = copy.deepcopy(a)
print(id(a)==id(e))
print(id(a[2])==id(e[2]))

# 相同的一个改变另一个也会随之改变

  22. 正则表达式

    常用于网络爬虫

    import re

p1 = 'cat'
p2 = 'bird'
string = 'dog runs to cat'
print(re.search(p1,string))
print(re.search(p2,string))

# [] 匹配多种可能
ptn =r"r[au]n"
print(re.search(ptn,string))

print(re.search(r"r[A-Z]n",string))
print(re.search(r"r[a-z]n",string))
print(re.search(r"r[0-9]n",string))
print(re.search(r"r[0-9 a-z]n",string))

# 数字
print(re.search(r"r\dn",string)) # \d 所有数字
print(re.search(r"r\Dn",string)) # \D 不是数字

# \s 所有空白符 如 \t \n
# \S 不是空白符
# \w 字母数字_  [a-zA-Z0-9_]
# \W 不是[a-zA-Z0-9_]的
# \b 一个空格
print(re.search(r"\bruns\b",'dog runs to cat'))
print(re.search(r"\bruns\b",'dog runsto cat'))
# \B 一个以上空格
print(re.search(r"\B runs \B",'dog  runs  to cat'))
print(re.search(r"\B runs \B",'dog runs to cat'))
# \\ 匹配 \
# .  匹配任何字符

# ^ 句首  $句尾
print(re.search(r"^dog",'dog runs to cat'))
print(re.search(r"cat$",'dog runs to cat'))
# ? 是否（0次或1次）
print(re.search(r"cat(t)?",'catt'))
print(re.search(r"cat(t)?",'cat'))
# 多行匹配
string = """
s    dog
I Assertion
"""
print(re.search(r"^I",string))
print(re.search(r"^I",string,flags=re.M))

# * 0或多次
print(re.search(r"ab*","a"))
print(re.search(r"ab*","abbb"))
# + 1或多次
print(re.search(r"ab+","a"))
print(re.search(r"ab+","abbb"))
# {n,m} 出现n到m次
print(re.search(r"ab{2,10}","a"))
print(re.search(r"ab{2,10}","abbb"))

# group组 取内容
match = re.search(r"(\d+),Date:(.+)","ID:0201,Date:Feb/02/2017")
print(match.group())
print(match.group(1))
print(match.group(2))

match = re.search(r"(?P<id>\d+),Date:(?P<date>.+)","ID:0201,Date:Feb/02/2017")
print(match.group('id'))
print(match.group('date'))

# 寻找所有
# findall
print(re.findall(r"r[ua]n","run ran ren"))
# | 或者
print(re.findall(r"(run|ran)","run ran ren"))

# re.sub() 替换
print(re.sub(r"r[ua]n",'swim',"run ran ren"))
# re.split() 分裂
print(re.split(r"[,;\.]","r,a;e.n"))
