3.2 读取数据

首先我们要了解什么是数据集，通常数据有数据和对应的label。一般有两种组织形式，第一种是同一种label的在同一个文件夹下，文件夹名即为对应的label，第二种方式是，每一个数据对应一个同名的txt，txt中写着该数据的label。

3.2.1 第一种组织形式读数据

通过路径和路径名，读取图片以及返回图片和label。

import os # 操作系统包

from torch.utils.data import Dataset # 读取数据集的抽象类
from PIL import Image # 读图片包

# 实现类重写方法
class MyData(Dataset):

    def __init__(self,root_dir,label_dir):
        self.root_dir =root_dir # 根目录
        self.label_dir=label_dir # label目录
        self.path = os.path.join(self.root_dir,self.label_dir) # 拼接地址
        self.img_path=os.listdir(self.path) # 将self.path目录下的文件名列表

    def __getitem__(self, idx):
        img_name = self.img_path[idx] # 根据下表选择对应图片名
        img_item_path = os.path.join(self.root_dir,self.label_dir,img_name) # 拼接图片路径
        img=Image.open(img_item_path)
        label=self.label_dir # label
        return img,label # 返回img,label

    def __len__(self):
        return len(self.img_path) # 数据集大小（数据个数）

root_dir='dataset/train'
ants_label_dir = 'ants'
bees_label_dir = 'bees'
ants_dataset = MyData(root_dir,ants_label_dir)
bees_dataset = MyData(root_dir, bees_label_dir)

train_dataset =ants_dataset + bees_dataset # +可以将两个数据集拼接 前面是ants_dataset 后面是bees_dataset
3.2.2 第二种组织形式读数据

与上一种方法相似，只需要增加一步，进入label文件夹中读取图片对应的txt得到其label即可。

import os

from torch.utils.data import Dataset
from PIL import Image

class MyData(Dataset):

    def __init__(self,root_dir,image_dir,label_dir):
        self.root_dir =root_dir
        self.image_dir=image_dir
        self.label_dir = label_dir
        self.path1 = os.path.join(self.root_dir,self.image_dir)
        self.path2 = os.path.join(self.root_dir, self.label_dir)
        self.img_path=os.listdir(self.path1)
        self.lab_path = os.listdir(self.path2)

    def __getitem__(self, idx):
        img_name = self.img_path[idx]
        lab_name = self.lab_path[idx]
        img_item_path = os.path.join(self.root_dir,self.image_dir,img_name)
        img=Image.open(img_item_path)
        label_item_path = os.path.join(self.root_dir,self.label_dir,lab_name)
        with open(label_item_path,'r') as f:
            label=f.read()
        return img,label

    def __len__(self):
        return len(self.img_path)

root_dir='dataset2/train'
ants_image_dir = 'ants_image'
bees_image_dir = 'bees_image'
ants_label_dir = 'ants_label'
bees_label_dir = 'bees_label'
ants_dataset = MyData(root_dir,ants_image_dir,ants_label_dir)
bees_dataset = MyData(root_dir, bees_image_dir,ants_label_dir)

train_dataset =ants_dataset+bees_dataset

  1. add_image方法
def add_image(self, tag, img_tensor, global_step=None, walltime=None, dataformats='CHW'):
    """Add image data to summary.

    Note that this requires the ``pillow`` package.

    Args:
        tag (string): Data identifier
        img_tensor (torch.Tensor, numpy.array, or string/blobname): Image data
        global_step (int): Global step value to record
        walltime (float): Optional override default walltime (time.time())
          seconds after epoch of event
    Shape:
        img_tensor: Default is :math:`(3, H, W)`. You can use ``torchvision.utils.make_grid()`` to
        convert a batch of tensor into 3xHxW format or call ``add_images`` and let us do the job.
        Tensor with :math:`(1, H, W)`, :math:`(H, W)`, :math:`(H, W, 3)` is also suitable as long as
        corresponding ``dataformats`` argument is passed, e.g. ``CHW``, ``HWC``, ``HW``.

可以看出，img_tensor参数只能接受torch.Tensor，numpy.array，string/blobname类型。同时要注意图片shape的组织形式，默认为CHW，即（通道，高，宽)。可以通过dataformats参数进行设置。

3.3.2 image与scalar展示的实现
from torch.utils.tensorboard import SummaryWriter # 导入SummaryWriter类
import numpy as np
from PIL import Image

# 设置writer的log文件夹位置
writer=SummaryWriter('logs')

# 图片路径
image_path = "dataset2/train/ants_image/0013035.jpg"
img_PIL = Image.open(image_path)
img_array = np.array(img_PIL) # 将PIL图片类型转为numpy 原因看img_array方法解析

3.4.2 Normalize

  1. 源码说明

    class Normalize(torch.nn.Module):
    """Normalize a tensor image with mean and standard deviation.
    This transform does not support PIL Image.
    Given mean: ``(mean[1],...,mean[n])`` and std: ``(std[1],..,std[n])`` for ``n``
    channels, this transform will normalize each channel of the input
    ``torch.*Tensor`` i.e.,
    ``output[channel] = (input[channel] - mean[channel]) / std[channel]``

    .. note::
        This transform acts out of place, i.e., it does not mutate the input tensor.

    Args:
        mean (sequence): Sequence of means for each channel.
        std (sequence): Sequence of standard deviations for each channel.
        inplace(bool,optional): Bool to make this operation in-place.

    """

  2. 示例

    # ToTensor
trans_totensor = transforms.ToTensor()
img_tensor = trans_totensor(img)
writer.add_image("ToTensor",img_tensor)

# Normalize
print(img_tensor[0][0][0])
# 三条信道的均值mean与标准差std
trans_norm = transforms.Normalize([0.5,0.5,0.5], [0.5,0.5,0.5])
img_norm = trans_norm(img_tensor)
print(img_norm[0][0][0])
writer.add_image("Normalize",img_norm,1)

writer.close()
3.4.3 Resize

  1. 源码说明

    class Resize(torch.nn.Module):
    """Resize the input image to the given size.
    If the image is torch Tensor, it is expected
    to have [..., H, W] shape, where ... means an arbitrary number of leading dimensions
    """
    def forward(self, img):
        """
        Args:
            img (PIL Image or Tensor): Image to be scaled.

        Returns:
            PIL Image or Tensor: Rescaled image.
        """
        return F.resize(img, self.size, self.interpolation, self.max_size, self.antialias)

    该类可以生成一个resize实例，可以修改图片尺寸，若输入为(H,W)则伸缩为对应尺寸，若只有一个数据如下compose示例中，则为将图片的短边伸缩为对应尺寸，长边等比放缩。

  2. 示例

    # Resize  新版本的Resize好像没有魔术方法 可以使用 trans_resize.forward(img) == trans_resize(img)
print(img.size)
trans_resize = transforms.Resize((512,512))
# img PIL -> resize -> img_resize PIL
img_resize = trans_resize(img)
# img_resize PIL -> totensor -> img_resize tensor
img_resize=trans_totensor(img_resize)
writer.add_image("Resize",img_resize,0)
print(img_resize)

writer.close()
3.4.4 Compose

  1. 源码说明

    class Compose:
    """Composes several transforms together. This transform does not support torchscript.
    Please, see the note below."""

    Compose可以将一系列操作组合在一起，有先后顺序。

  2. 示例

    # Compose - resize - 2
trans_resize_2 = transforms.Resize(512)
# PIL -> PIL -> tensor  此处参数不能换位置 因为resize只能接收PIL对象
trans_compose = transforms.Compose([trans_resize_2, trans_totensor])
img_resize_2 = trans_compose(img)
writer.add_image("Resize",img_resize_2,1)

# RandomCrop 在图片中随机裁剪对应尺寸的图片
trans_random = transforms.RandomCrop((500, 1000))
trans_compose_2 = transforms.Compose([trans_random, trans_totensor])
for i in range(10):
    img_crop = trans_compose_2(img)
    writer.add_image("RandomCrop",img_crop,i)
    
writer.close()
