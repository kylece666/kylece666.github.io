3.2 读取数据

首先我们要了解什么是数据集，通常数据有数据和对应的label。一般有两种组织形式，第一种是同一种label的在同一个文件夹下，文件夹名即为对应的label，第二种方式是，每一个数据对应一个同名的txt，txt中写着该数据的label。

3.2.1 第一种组织形式读数据

通过路径和路径名，读取图片以及返回图片和label。

import os # 操作系统包

from torch.utils.data import Dataset # 读取数据集的抽象类
from PIL import Image # 读图片包

# 实现类重写方法
class MyData(Dataset):

    def __init__(self,root_dir,label_dir):
        self.root_dir =root_dir # 根目录
        self.label_dir=label_dir # label目录
        self.path = os.path.join(self.root_dir,self.label_dir) # 拼接地址
        self.img_path=os.listdir(self.path) # 将self.path目录下的文件名列表

    def __getitem__(self, idx):
        img_name = self.img_path[idx] # 根据下表选择对应图片名
        img_item_path = os.path.join(self.root_dir,self.label_dir,img_name) # 拼接图片路径
        img=Image.open(img_item_path)
        label=self.label_dir # label
        return img,label # 返回img,label

    def __len__(self):
        return len(self.img_path) # 数据集大小（数据个数）

root_dir='dataset/train'
ants_label_dir = 'ants'
bees_label_dir = 'bees'
ants_dataset = MyData(root_dir,ants_label_dir)
bees_dataset = MyData(root_dir, bees_label_dir)

train_dataset =ants_dataset + bees_dataset # +可以将两个数据集拼接 前面是ants_dataset 后面是bees_dataset
3.2.2 第二种组织形式读数据

与上一种方法相似，只需要增加一步，进入label文件夹中读取图片对应的txt得到其label即可。

import os

from torch.utils.data import Dataset
from PIL import Image

class MyData(Dataset):

    def __init__(self,root_dir,image_dir,label_dir):
        self.root_dir =root_dir
        self.image_dir=image_dir
        self.label_dir = label_dir
        self.path1 = os.path.join(self.root_dir,self.image_dir)
        self.path2 = os.path.join(self.root_dir, self.label_dir)
        self.img_path=os.listdir(self.path1)
        self.lab_path = os.listdir(self.path2)

    def __getitem__(self, idx):
        img_name = self.img_path[idx]
        lab_name = self.lab_path[idx]
        img_item_path = os.path.join(self.root_dir,self.image_dir,img_name)
        img=Image.open(img_item_path)
        label_item_path = os.path.join(self.root_dir,self.label_dir,lab_name)
        with open(label_item_path,'r') as f:
            label=f.read()
        return img,label

    def __len__(self):
        return len(self.img_path)

root_dir='dataset2/train'
ants_image_dir = 'ants_image'
bees_image_dir = 'bees_image'
ants_label_dir = 'ants_label'
bees_label_dir = 'bees_label'
ants_dataset = MyData(root_dir,ants_image_dir,ants_label_dir)
bees_dataset = MyData(root_dir, bees_image_dir,ants_label_dir)

train_dataset =ants_dataset+bees_dataset

3.4.2 Normalize

  1. 源码说明

    class Normalize(torch.nn.Module):
    """Normalize a tensor image with mean and standard deviation.
    This transform does not support PIL Image.
    Given mean: ``(mean[1],...,mean[n])`` and std: ``(std[1],..,std[n])`` for ``n``
    channels, this transform will normalize each channel of the input
    ``torch.*Tensor`` i.e.,
    ``output[channel] = (input[channel] - mean[channel]) / std[channel]``

    .. note::
        This transform acts out of place, i.e., it does not mutate the input tensor.

    Args:
        mean (sequence): Sequence of means for each channel.
        std (sequence): Sequence of standard deviations for each channel.
        inplace(bool,optional): Bool to make this operation in-place.

    """

  2. 示例

    # ToTensor
trans_totensor = transforms.ToTensor()

writer.add_image("ToTensor",img_tensor)

# Normalize
print(img_tensor[0][0][0])
# 三条信道的均值mean与标准差std
trans_norm = transforms.Normalize([0.5,0.5,0.5], [0.5,0.5,0.5])

print(img_norm[0][0][0])
writer.add_image("Normalize",img_norm,1)

3.4.3 Resize

  1. 源码说明

    class Resize(torch.nn.Module):
    """Resize the input image to the given size.
    If the image is torch Tensor, it is expected
    to have [..., H, W] shape, where ... means an arbitrary number of leading dimensions
    """
    def forward(self, img):
        """
        Args:
            img (PIL Image or Tensor): Image to be scaled.

        Returns:
            PIL Image or Tensor: Rescaled image.
        """
        return F.resize(img, self.size, self.interpolation, self.max_size, self.antialias)

    该类可以生成一个resize实例，可以修改图片尺寸，若输入为(H,W)则伸缩为对应尺寸，若只有一个数据如下compose示例中，则为将图片的短边伸缩为对应尺寸，长边等比放缩。

import torchvision

img,target = test_set[0]

4.2 卷积

  • 参数解析

    • weight：卷积核
    • stride：一次走几步
    • padding：在input周围包一圈0，若输入为单一数字则上下左右包n层，若元组(H,W)，则上下包H层左右包W层
[1, 2, 0, 3, 1]               
[0, 1, 2, 3, 1]               
[1, 2, 1, 0, 0]      [1, 2, 1]      [10, 12, 12]    
[5, 2, 3, 1, 1]      [0, 1, 0]      [18, 16, 16]
[2, 1, 0, 1, 1]      [2, 1, 0]      [13,  9,  3]
     input             kernel           output

卷积：对应位置相乘求和，如10 = 1x1+2x2+0x1+0x0+1x1+2x0+1x2+2x1+1x0=1+4+0+0+1+0+2+2+0 = 10

import torch
import torch.nn.functional as F

input = torch.tensor([[1, 2, 0, 3, 1],
                      [0, 1, 2, 3, 1],
                      [1, 2, 1, 0, 0],
                      [5, 2, 3, 1, 1],
                      [2, 1, 0, 1, 1]])

kernel = torch.tensor([[1, 2, 1],
                       [0, 1, 0],
                       [2, 1, 0]])

input = torch.reshape(input, (1, 1, 5, 5))
kernel = torch.reshape(kernel, [1, 1, 3, 3])

print(input.shape)
print(kernel.shape)

output = F.conv2d(input, kernel, stride=1)
print(output)

output2 = F.conv2d(input, kernel, stride=2)
print(output2)

output3 = F.conv2d(input, kernel, stride=1, padding=1)
print(output3)

4.3 卷积层

官方文档：Conv2d — PyTorch 1.8.1 documentation

  • 参数：
    • in_channels: 输入通道数
    • out_channels: 输出通道数，输出的通道数等于卷积核的个数
    • kernel_size: 卷积核尺寸
    • stride: 卷积核移动的步长
    • padding: 在周围进行填充

示例：

import torch
import torchvision
from torch import nn
from torch.utils.data import DataLoader
from torch.nn import Conv2d
from torch.utils.tensorboard import SummaryWriter

dataset = torchvision.datasets.CIFAR10("./dataset3",train=False,transform=torchvision.transforms.ToTensor(),
                                       download=True)
dataloader = DataLoader(dataset,batch_size=64)


class Mo(nn.Module):
    def __init__(self) -> None:
        super().__init__()
        # 输入channel为3，输出为6
        self.conv1=Conv2d(in_channels=3,out_channels=6,kernel_size=3,stride=1,padding=0)

    def forward(self,input):
        output = self.conv1(input)
        return output

mo = Mo()
print(mo)

writer = SummaryWriter("./logs")

step = 0
for data in dataloader:
    imgs,targets=data
    output = mo(imgs)
    print(imgs.shape)
    print(output.shape)
    # torch.Size([64, 3, 32, 32])
    writer.add_images("input",imgs,step)
    # torch.Size([64, 6, 30, 30]) -> [xxx, 3, 30, 30]
    # 由于tensorBoard无法显示6个channel 对output进行reshape -1的意思是根据后面三个参数进行自动计算
    # 由于本来本来每组64张3信道，现在要将6信道拆成两个三信道，每组显示128张图片
    output =torch.reshape(output,(-1,3,30,30))
    writer.add_images("output", output, step)
    step +=1

writer.close()

4.4 池化层

官方文档：torch.nn — PyTorch 1.8.1 documentation

  • 参数

    • kernel_size: 卷积核大小

    • stride: 移动步长默认为kernel_size

    • padding: 周围填充

    • dilation: 空洞卷积

    • ceil_mode: 当移动到边缘时，不足kernel_size x kernel_size个元素时，是否计算，ceil为保留计算，floor为不计算。

      如：

      [1, 2, 0, 3, 1]               
[0, 1, 2, 3, 1]               
[1, 2, 1, 0, 0]    
[5, 2, 3, 1, 1]     [2,3]
[2, 1, 0, 1, 1]     [5,1]       [2]
     input           True      False

池化层的作用是，提取特征值，在保留原始图像的特征的同时，缩小图像尺寸

import torch
import torchvision
from torch import nn
from torch.nn import Linear, Conv2d, Sequential, MaxPool2d, Flatten
from torch.utils.data import DataLoader

dataset = torchvision.datasets.CIFAR10("./dataset3",train=False,transform=torchvision.transforms.ToTensor(),
                                       download=True)
dataloader = DataLoader(dataset,batch_size=64)


class Module(nn.Module):
    def __init__(self) -> None:
        super().__init__()
        self.model1 = Sequential(
            Conv2d(3,32,5,padding=2),
            MaxPool2d(2),
            Conv2d(32,32,5,padding=2),
            MaxPool2d(2),
            Conv2d(32,64,5,padding=2),
            MaxPool2d(2),
            Flatten(),
            Linear(1024,64),
            Linear(64,10)
        )

    def forward(self,input):
        output = self.model1(input)
        return output

loss = nn.CrossEntropyLoss()
model = Module()

for epoch in range(20):

5.2 训练模型及保存

"""
train.py
"""
import torch
import torchvision
from torch.utils.tensorboard import SummaryWriter

from model import *  # 与model.python放在同级目录 导入模型
from torch.utils.data import DataLoader

# 训练集
train_data = torchvision.datasets.CIFAR10(root="../dataset3", train=True, transform=torchvision.transforms.ToTensor(),
                                      download=True)
# 测试集
test_data = torchvision.datasets.CIFAR10(root="../dataset3", train=False, transform=torchvision.transforms.ToTensor(),
                                      download=True)

# length 长度
train_data_size = len(train_data)
test_data_size = len(test_data)
print("训练集长度：{}".format(train_data_size))
print("测试集长度：{}".format(test_data_size))

# DataLoader 加载数据集
train_loader = DataLoader(train_data, batch_size=64)
test_loader = DataLoader(test_data, batch_size=64)

# 创建网络模型
model = Model()

# 损失函数
loss_fn = nn.CrossEntropyLoss()

# 优化器
learning_rate = 1e-2
optimizer = torch.optim.SGD(model.parameters(),lr=learning_rate)

# 设置训练网络的一些参数

# 记录训练次数
total_train_step = 0
# 记录测试次数
total_test_step = 0
# 训练的轮数
epoch = 50

# 添加tensorboard
writer = SummaryWriter("./logs_train")

for i in range(epoch):
    print("-----------第{}轮训练开始-----------".format(i+1))

    # 训练步骤开始
    model.train() # 并不是有此句才表示模型训练，只是在包含某些层时会修改参数，可以查看官网
    for data in train_loader:
        imgs, targets = data
        output = model(imgs)
        loss = loss_fn(output,targets)

        # 优化器优化模型
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        total_train_step += 1
        if total_train_step % 100 == 0:
            print("训练次数：{}，Loss：{}".format(total_train_step,loss.item()))
            writer.add_scalar("train_loss",loss.item(),total_train_step)

    # 测试步骤开始
    total_test_loss = 0
    total_accuracy = 0
    model.eval()
    with torch.no_grad():
        for data in test_loader:
            imgs,targets = data
            output = model(imgs)
            loss=loss_fn(output,targets)
            total_test_loss = total_test_loss + loss.item()
            # argmax(1)表示在行的方向上取最大值下标 == 产生[T,F,T,F,T,...]数组 .sum()对其求和T1 F0
            accuracy = (output.argmax(1) == targets).sum()
            total_accuracy += accuracy
    print("整体测试集上的Loss：{}".format(total_test_loss))
    print("整体测试集上的正确率：{}".format(total_accuracy/test_data_size))
    writer.add_scalar("test_loss", total_test_loss, total_test_step)
    writer.add_scalar("test_accuracy", total_accuracy/test_data_size, total_test_step)
    total_test_step += 1

    torch.save(model,"model_{}.pth".format(i))
    # torch.save(model.state_dict(),"model_{}.pth".format(i))
    print("模型已保存")

writer.close()
